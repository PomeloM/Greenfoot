import greenfoot.*;  // (World, Actor, GreenfootImage, Greenfoot and MouseInfo)
import java.util.List;

/**
 * Write a description of class Block here.
 * 
 * @author (your name) 
 * @version (a version number or a date)
 */
public class Block extends Actor
{
    private boolean bomb = false;
    private boolean flag = false;

    public void setBomb() {
        bomb = true;
    }

    public boolean getBomb() {
        return bomb;
    }

    /**
     * Act - do whatever the Block wants to do. This method is called whenever
     * the 'Act' or 'Run' button gets pressed in the environment.
     */
    public void act() 
    {
        // Add your action code here.
        if (Greenfoot.mouseClicked(this)) {
            MouseInfo mouse = Greenfoot.getMouseInfo();
            //按下右鍵
            if (mouse.getButton()==3) {
                if (flag) {                    
                    setImage("Block.png");
                } else {
                    setImage("BlockFlagged.png");
                }
                flag = !flag;
            } else {
                //沒有被標旗子才能按左鍵打開
                if (!flag) {
                    if (bomb) {
                        setImage("Bomb.png");
                        Greenfoot.playSound("bomb.wav");
                    } else {
                        setImage("BlockClicked["+getBombNumber()+"].png");
                    }
                }
            }            
        }
    }    

    public int getBombNumber() {
        int bombNumber=0;
        List<Block> blocks = getNeighbours(1, true, Block.class);
        for (Block block: blocks) {
            if (block.getBomb()) {
                bombNumber++;
            }
        }       
        return bombNumber;        
    }
}

  • 案例研究：踩地雷：任務六
    • 任務六：打開後的磚塊不能再插旗子。
  • 案例研究：踩地雷：任務七
    • 任務七：若是開到炸彈，則顯示所有炸彈，然後遊戲結束。
  • 案例研究：踩地雷：任務八
    • 任務八：若是該磚塊周圍的磚塊的炸彈數是零，則再展開該磚塊。(遞迴)
  • 案例研究：踩地雷：任務九
    • 任務九：檢查10個炸彈磚塊是否都被標上旗子，若是，則播放音樂，遊戲停止。
