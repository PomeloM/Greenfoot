import greenfoot.*;  // (World, Actor, GreenfootImage, and Greenfoot)

public class Key extends Actor
{
    /**
     * Create a new key.
     */
    public Key()
    {
    }

    /**
     * 每個回合執行的動作。
     */
    public void act()
    {
        // 如果按下 g 鍵，圖片切換白鍵按下圖，並播放中音 Sol 。
        if (Greenfoot.isKeyDown("g")){
            setImage("white-key-down.png");
            Greenfoot.playSound("3g.wav");
        } else { // 否則（放開g鍵），圖片切回白鍵原圖。
            setImage("white-key.png");
        }
    }

}

import greenfoot.*;  // (World, Actor, GreenfootImage, and Greenfoot)

public class Key extends Actor
{

    // 按鍵有沒有按下。
    private boolean isDown;    

    /**
     * Create a new key.
     */
    public Key()
    {
    }

    /**
     * 每個回合執行的動作。
     */
    public void act()
    {
        // 如果g鍵沒按下，這時按下g鍵。
        if  ( !isDown && Greenfoot.isKeyDown("g"))  {
            setImage("white-key-down.png");
            Greenfoot.playSound("3g.wav");
            isDown = true;   // 記錄g鍵按下了。
        }
        // 如果g鍵按下了，這時放開g鍵。
        if  ( isDown && !Greenfoot.isKeyDown("g"))  {
            setImage("white-key.png");
            isDown = false;  // 記錄g鍵放開了。
        } 
    }
}

為了解決當放開g鍵時，setImage和playSound只執行一次，而非反覆執行，在Key class中定義一個資料成員isDown，用於記錄琴鍵目前是否處於被按下的狀態：

被按下時是true

放開時是false

我們必需判斷以下兩個情況

  1. 琴鍵處於正常狀態而使用者按下g鍵：

使用者剛按下g鍵

  1. 琴鍵處於按下狀態而使用者未按g鍵：

使用者剛釋放g鍵

2.產生多個琴鍵，並指定到不同的音效。¶

  1. 目前這個專案的問題：按下鍵盤上的g後，所有的琴鍵都做出相同的回應。

問題的原因：程式中將對應的鍵盤按鍵及所要播放的聲音檔寫死。

  1. 抽象化(Abstraction)的程式設計概念: 程式不只是用於解決特定問題，而是要能夠解決一般性的問題。

例如：我們不能設計一個僅能找1764的平方根程式，而應設計可找任意數平方根的程式。

同樣地，我們不能設計一個只能play 「a」sound的程式，而應設計一個可以play各個音階的程式。

  1. 在抽象化這種設計概念之下，不同的Key物件需對應不同的鍵盤按鍵；同時，不同的Key物件被按下時，要能播放對應音階。

程式作法：

在Key類別中定義兩個資料成員，分別用於記錄Key物件所要回應的鍵盤按鍵，以及被按下時所要播放的聲音檔。

型態：String

名稱：key及sound

在Key 類別中定義兩個資料成員。

import greenfoot.*;  // (World, Actor, GreenfootImage, and Greenfoot)

public class Key extends Actor
{
    // 鍵盤按鍵
    private String key;
    // 聲音檔
    private String sound;

    // 按鍵有沒有按下。
    private boolean isDown;     

    /** 琴鍵放開的圖片。 */
    private String upImage;
    /** 琴鍵按下的圖片。 */
    private String downImage;

    /**
     * 建立新的琴鍵。
     *
     * @param keyName 琴鍵的鍵盤按鍵
     * @param soundFile 琴鍵的音效檔
     * @param img1琴鍵放開的圖片
     * @param img2琴鍵按下的圖片
     */
    public Key(String keyName, String soundFile, String img1, String img2)
    {
        key = keyName;
        sound = soundFile+".wav";
        upImage = img1;
        downImage = img2;
        setImage(upImage);
    }

    /**
     * 每個回合執行的動作。
     */
    public void act()
    {
        // 如果鍵盤按鍵沒按下，這時按下鍵盤按鍵。
        if ( !isDown && Greenfoot.isKeyDown(key)) {
            setImage(downImage);
            Greenfoot.playSound(sound);
            isDown = true;   // 記錄鍵盤按鍵按下了。
        }
        // 如果鍵盤按鍵按下了，這時放開鍵盤按鍵。
        if ( isDown && !Greenfoot.isKeyDown(key)) {
            setImage(upImage);
            isDown = false;  // 記錄鍵盤按鍵放開了。
        } 
    }
}

import greenfoot.*;  // (World, Actor, GreenfootImage, and Greenfoot)

/**
 * A piano that can be played with the computer keyboard.
 * 
 * @author: M. Kolling
 * @version: 0.1
 */
public class Piano extends World
{
    /** 白鍵的鍵盤按鍵。 */
    private String[] whiteKeys = {"a","s","d","f","g","h","j","k","l",";","'","/"};
    /** 白鍵的音符。 */
    private String[] whiteNotes = {"3c","3d","3e","3f","3g","3a","3b","4c","4d","4e","4f","4g"};

    /** 黑鍵的鍵盤按鍵。 */
    private String[] blackKeys =  { "W", "E", "", "T", "Y", "U", "", "O", "P", "", "]" };
    /** 黑鍵的音符。 */
    private String[] blackNotes = { "3c#", "3d#", "", "3f#", "3g#", "3a#", "", "4c#", "4d#", "", "4f#" };

    /**
     * Make the piano.
     */
    public Piano() 
    {
        super(800, 340, 1);

        // 用迴圈加上12個相同的白鍵。
        for (int i=0; i<whiteKeys.length; i++) { 
            // 加上白鍵，X座標在54+i*63，Y座標在180。
            addObject(new Key(whiteKeys[i],whiteNotes[i]+".wav","white-key.png","white-key-down.png"), 54+i*63, 180);
        }

        // 建立鋼琴的黑鍵。
        for (int i = 0; i < blackKeys.length; i++) {
            // 如果是空字串，表示這一格沒有黑鍵，要跳過。
            if ( ! blackKeys[i].equals("") ) {
                // 建立黑鍵，聲音檔名是「音符.wav」。
                Key key = new Key(blackKeys[i], blackNotes[i]+".wav", "black-key.png", "black-key-down.png");
                // 加上黑鍵，X座標在85+(i*63)，Y座標在125。
                addObject(key, 85 + (i*63), 125);
            }
        }
    }
}
